    常用命令#

    vim删除文件中的空行#

    :g/^$/d

    awk删除某一列#

    #删除第一列
awk '{$1=null;print $0}' filename

    awk中匹配正则表达式#

    #输出第一列匹配到scaffold*的行
awk '{if($1~/scaffold*/) print $0}' filename

    删除字符串中的换行符#

    echo $var |tr -d '\n\r'

    免输入密码进行scp传输#

    #-p 指定密码
sshpass -p '123456' scp root@host_ip:/home/test/t ./tmp/
